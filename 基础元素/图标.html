            searchIcon(e) {
                let key = e.detail.value.toLowerCase();

                for (let i = 0; i < list.length; i++) {
                    let a = key;
                    let b = list[i].name.toLowerCase();

                        list[i].isShow = true
                    } else {
                        list[i].isShow = false
